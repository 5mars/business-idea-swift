**Step 4: Modify `confirmCompletion(id:outcome:note:)` method** to set celebration state AFTER setting `justCompletedActionId`. Insert this block after the `justCompletedActionId = id` line and its 0.6s clear:

```swift
// Celebration state (Phase 3)
let newCompletedCount = microActions.filter(\.isCompleted).count
let allDone = newCompletedCount == microActions.count

if allDone {
    // Plan complete takes priority — even if count is 3, 5, or 7
    celebrationState = .planComplete
} else if [3, 5, 7].contains(newCompletedCount) {
    celebrationState = .milestone(count: newCompletedCount)
    // Auto-clear after 2.5s
    DispatchQueue.main.asyncAfter(deadline: .now() + 2.5) { [weak self] in
        if self?.celebrationState == .milestone(count: newCompletedCount) {
            self?.celebrationState = .idle
        }
    }
} else {
    celebrationState = .inlineConfetti(actionId: id)
    // Auto-clear after 1.5s
    DispatchQueue.main.asyncAfter(deadline: .now() + 1.5) { [weak self] in
        if self?.celebrationState == .inlineConfetti(actionId: id) {
            self?.celebrationState = .idle
        }
    }
}
```
