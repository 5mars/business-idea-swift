struct ActionPlanDetailView: View {
    let planId: UUID

    @StateObject private var viewModel = ActionPlanViewModel()
    @State private var selectedAction: MicroAction? = nil
    @State private var showCommitmentSheet = false

            .presentationBackground(Color.appBg)

Modify JourneyNodeView to add completion bounce and unlock animations.

**Add parameters to JourneyNodeView init:**
- `justCompletedActionId: UUID?` — from viewModel, used to detect if this node's predecessor was just completed
- `index: Int` — this node's index in the microActions array
- `actions: [MicroAction]` — full array, needed to find predecessor

**Add local animation state:**
```swift
@State private var isAnimatingCompletion = false
@State private var unlockAnimating = false
```

**Scale bounce on completion:**

Apply `.scaleEffect(isAnimatingCompletion ? 1.2 : 1.0)` to the circle ZStack.

Add an `.onChange(of: state)` modifier:
```swift
.onChange(of: state) { oldValue, newValue in
    if oldValue != .completed && newValue == .completed {
        // Bounce up
        AnimationPolicy.animate(.spring(response: 0.15, dampingFraction: 0.4)) {
            isAnimatingCompletion = true
        }
        // Bounce back
        DispatchQueue.main.asyncAfter(deadline: .now() + 0.15) {
            AnimationPolicy.animate(.spring(response: 0.15, dampingFraction: 0.6)) {
                isAnimatingCompletion = false
            }
        }
    }
}
```

**Unlock animation on successor node:**

```swift
.onChange(of: justCompletedActionId) { _, completedId in
    guard let completedId = completedId,
          let completedIndex = actions.firstIndex(where: { $0.id == completedId }),
          index == completedIndex + 1 else { return }
    // This node is the successor — animate unlock
    AnimationPolicy.animate(.spring(response: 0.5, dampingFraction: 0.7)) {
        unlockAnimating = true
    }
    DispatchQueue.main.asyncAfter(deadline: .now() + 0.5) {
        unlockAnimating = false
    }
}
```

Apply to the circle: `.scaleEffect(unlockAnimating ? 1.15 : 1.0)` (a subtle pulse when unlocking).

The combined scale effect on the circle ZStack should multiply both:
```swift
.scaleEffect(isAnimatingCompletion ? 1.2 : 1.0)
.scaleEffect(unlockAnimating ? 1.15 : 1.0)
```

**Update JourneyPathView.swift** to pass the new parameters to JourneyNodeView:

JourneyNodeView(

**Important:** All animation calls MUST go through `AnimationPolicy.animate { }` — never call `withAnimation` directly.
